    name: zynochat-site
    runtime: static
    buildCommand: ""
    staticPublishPath: .
    pullRequestPreviewsEnabled: false
    headers:

        name: X-Frame-Options
        value: DENY

        name: X-Content-Type-Options
        value: nosniff

        value: strict-origin-when-cross-origin

        value: "1; mode=block"

        name: Cache-Control
        value: public, max-age=2592000
    routes:

        source: /
        destination: /index.html

        destination: /
